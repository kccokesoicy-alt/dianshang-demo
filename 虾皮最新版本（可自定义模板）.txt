class Utils:
    @staticmethod
    def setup_logger(name, filename):
        log_path = Path(Config.RESULT_DIR) / "logs" / filename
        log_path.parent.mkdir(parents=True, exist_ok=True)
        logger = logging.getLogger(name)
        logger.setLevel(logging.INFO)
        if not logger.handlers:
            fh = logging.FileHandler(log_path, encoding='utf-8')
            ch = logging.StreamHandler()
            formatter = logging.Formatter('%(asctime)s - %(message)s')
            fh.setFormatter(formatter)
            ch.setFormatter(formatter)
            logger.addHandler(fh)

    @staticmethod
    def retry(times=3, delay=2):
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                for i in range(times):
                    try:
                        return func(*args, **kwargs)
                    except Exception as e:
                        if i == times - 1: raise e
                        time.sleep(delay)
                return None

            return wrapper

        return decorator

    @staticmethod
    def atomic_save(df_or_wb, path):
        # 【防呆防御】强制在保存的最后一刻检查并创建目标文件夹！
        path_obj = Path(path)
        path_obj.parent.mkdir(parents=True, exist_ok=True)

        path_str = str(path)
        temp_path = path_str + ".temp.xlsx"
        try:
            if isinstance(df_or_wb, pd.DataFrame):
                df_or_wb.to_excel(temp_path, index=False, engine='openpyxl')
            else:
                df_or_wb.save(temp_path)
            if os.path.exists(path_str): os.remove(path_str)
            shutil.move(temp_path, path_str)
        except Exception as e:
            print(f"❌ 文件保存失败: {e}")

    @staticmethod
    def robust_sku_clean(sku):
        if pd.isna(sku): return ""
        sku = str(sku).strip()
        sku = unicodedata.normalize('NFC', sku)
        sku = re.sub(r'[\u2000-\u200f\u2028-\u202f\u205f\u3000\ufeff]', '', sku)
        return sku.strip()

    @staticmethod
    def parse_erp_full_info(text):
        if not isinstance(text, str): return "", ""
        name_match = re.search(r"品名[:：]\s*(.*?)\s*(?:/|$)", text, re.IGNORECASE)
        clean_name = name_match.group(1).strip() if name_match else ""
        desc_suffix = ""
        slash_index = text.find('/')
        if slash_index != -1:
            raw_attrs = text[slash_index:]
            for stop_char in ['【', '[']:
                stop_idx = raw_attrs.find(stop_char)
                if stop_idx != -1: raw_attrs = raw_attrs[:stop_idx]
            parts = [p.strip() for p in raw_attrs.split('/') if p.strip()]
            desc_suffix = "\n".join(parts)
        return clean_name, desc_suffix

    @staticmethod
    def extract_erp_name(text):
        name, _ = TitleAnalyzer.parse_erp_full_info(text)
        return name

    @staticmethod
    def is_same_series(t1, t2):
        def clean(t):
            t = re.sub(r'[^\w\s]', '', str(t).lower())
            for w in TitleAnalyzer.NOISE_WORDS: t = t.replace(w, '')
            return t.strip()

        c1, c2 = clean(t1), clean(t2)
        if c1 == c2 and c1 != "": return True
        return difflib.SequenceMatcher(None, t1, t2).ratio() > 0.85

    @staticmethod
    def _extract_attributes_from_text(text):
        found_attrs = []
        text_lower = text.lower()
        for category, keywords in TitleAnalyzer.ATTR_LIB.items():
            for kw in keywords:
                pattern = r'\b' + re.escape(kw.lower()) + r'\b'
                if re.search(pattern, text_lower):
                    found_attrs.append(kw)
                    break
        size_match = re.search(r'\b(us\s*)?([4-9]|1[0-5])\b', text_lower)
        if size_match: found_attrs.append(f"Size {size_match.group(2)}")
        return found_attrs

    @staticmethod
    def generate_smart_title_data(titles):
        titles = sorted(list(set([str(t).strip() for t in titles if str(t).strip() and str(t).lower() != 'nan'])))
        if not titles: return "Fashion Jewelry Accessories", {}
        if len(titles) == 1: return titles[0], {titles[0]: "One Style"}

        prefix = os.path.commonprefix(titles)
        if ' ' in prefix and not prefix.endswith(' '): prefix = prefix[:prefix.rfind(' ') + 1]
        remainders = [t[len(prefix):] for t in titles]
        suffix = os.path.commonprefix([r[::-1] for r in remainders])[::-1]
        if ' ' in suffix and not suffix.startswith(' '): suffix = suffix[suffix.find(' '):]

        diff_map = {}
        all_keywords = []

        for idx, t in enumerate(titles):
            if len(t) >= len(prefix) + len(suffix):
                raw_diff = t[len(prefix):len(t) - len(suffix)].strip()
            else:
                raw_diff = t.replace(prefix, "").replace(suffix, "").strip()

            detected_attrs = TitleAnalyzer._extract_attributes_from_text(raw_diff)
            if detected_attrs:
                clean_diff = " ".join(detected_attrs)
            else:
                clean_diff = re.sub(r'[^\w\s]', '', raw_diff).strip().title()
                if not clean_diff or (clean_diff.isdigit() and len(clean_diff) < 3):
                    clean_diff = f"Style {chr(65 + idx)}" if idx < 26 else f"Style {idx + 1}"

            diff_map[t] = clean_diff
            if "Style " not in clean_diff: all_keywords.append(clean_diff)

        unique_kws = sorted(list(set(all_keywords)), key=all_keywords.index)
        middle_part = " ".join(unique_kws[:3])
        final_title = f"{prefix} {middle_part} {suffix}".strip()
        final_title = re.sub(r'\s+', ' ', final_title)

        longest_orig_title = max(titles, key=len)
        if len(final_title) < 10 or len(final_title) < len(longest_orig_title) * 0.5:
            final_title = longest_orig_title

        return final_title, diff_map


class VisualEngine:
    @staticmethod
    def normalize_url(url):
        if not url: return ""
        return url.split("?")[0].replace("_60x60.jpg", ".jpg").replace("_60x60.jpeg", ".jpeg").strip()

    @staticmethod
    def download_bytes(url, cookies=None):
        try:
            headers = {"User-Agent": Config.USER_AGENT, "Referer": "https://admin.rinhoo1688.com/"}
            resp = requests.get(VisualEngine.normalize_url(url), headers=headers, cookies=cookies, verify=False,
                                timeout=5)
            return resp.content if resp.status_code == 200 else None
        except:
            return None

    @staticmethod
    def bytes_to_cv2(b):
        if not b: return None
        return cv2.imdecode(np.asarray(bytearray(b), dtype=np.uint8), cv2.IMREAD_COLOR)

    @staticmethod
    def calc_dhash(image, hash_size=8):
        try:
            gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
            resized = cv2.resize(gray, (hash_size + 1, hash_size))
            diff = resized[:, 1:] > resized[:, :-1]
            return sum([2 ** i for (i, v) in enumerate(diff.flatten()) if v])
        except:
            return 0

    @staticmethod
    def calc_hist_sim(img1, img2):
        if img1 is None or img2 is None: return 0.0
        try:
            h1 = cv2.calcHist([cv2.cvtColor(img1, cv2.COLOR_BGR2HSV)], [0, 1], None, [60, 32], [0, 180, 0, 256])
            h2 = cv2.calcHist([cv2.cvtColor(img2, cv2.COLOR_BGR2HSV)], [0, 1], None, [60, 32], [0, 180, 0, 256])
            cv2.normalize(h1, h1, 0, 1, cv2.NORM_MINMAX)
            cv2.normalize(h2, h2, 0, 1, cv2.NORM_MINMAX)
            return cv2.compareHist(h1, h2, cv2.HISTCMP_CORREL)
        except:
            return 0.0

    @staticmethod
    def is_duplicate(new_img, seen_fps):
        nh = VisualEngine.calc_dhash(new_img)
        for item in seen_fps:
            if bin(int(nh) ^ int(item['hash'])).count('1') <= 5:
                if VisualEngine.calc_hist_sim(new_img, item['img']) > 0.9: return True
        return False

    @staticmethod
    def classify(cv2_img):
        if cv2_img is None: return "unknown"
        try:
            gray = cv2.cvtColor(cv2_img, cv2.COLOR_BGR2GRAY)
            h, w = gray.shape

            borders = [gray[0:10, :], gray[h - 10:h, :], gray[:, 0:10], gray[:, w - 10:w]]
            border_mean = np.mean([np.mean(b) for b in borders])
            is_white_bg = border_mean > 210

            edges = cv2.Canny(gray, 50, 150)
            edge_density = cv2.countNonZero(edges) / (h * w)

            if is_white_bg:
                if edge_density > 0.045:
                    return "size"
                else:
                    return "white_bg"

            hsv = cv2.cvtColor(cv2_img, cv2.COLOR_BGR2HSV)
            skin_mask = cv2.inRange(hsv, np.array([0, 20, 70]), np.array([20, 255, 255]))
            if cv2.countNonZero(skin_mask) / (h * w) > 0.08:
                return "model"

            return "placement"
        except:
            return "unknown"

class DataMerger:
    @staticmethod
    def preprocess_files(root_folder):
        print(f"📂 扫描: {root_folder}")
        all_dfs = []
        for root, _, files in os.walk(root_folder):
            for file in files:
                path = Path(root) / file
                if path.suffix.lower() in ['.csv', '.xlsx', '.xls']:
                    try:
                        df = pd.read_csv(path) if path.suffix == '.csv' else pd.read_excel(path)
                        sku_col, title_col = None, None
                        df.columns = [str(c).strip() for c in df.columns]
                        for c in df.columns:
                            c_lower = str(c).lower()
                            if 'sku' in c_lower or '货号' in c_lower: sku_col = c
                            if '标题' in c_lower or 'title' in c_lower or '品名' in c_lower: title_col = c

                        if sku_col is None: sku_col = df.columns[0]
                        if title_col is None: title_col = df.columns[1] if len(df.columns) > 1 else df.columns[0]

                        df_clean = df[[sku_col, title_col]].copy()
                        df_clean.columns = ['SKU', 'Title']
                        df_clean['SKU'] = df_clean['SKU'].apply(Utils.robust_sku_clean)
                        df_clean['Title'] = df_clean['Title'].astype(str).str.strip()
                        df_clean = df_clean[df_clean['SKU'] != 'nan']
                        df_clean = df_clean[df_clean['SKU'] != '']
                        df_clean = df_clean[~df_clean['SKU'].str.lower().isin(['sku', '货号'])]
                        all_dfs.append(df_clean)
                    except Exception as e:
                        print(f"⚠️ 读取文件 [{file}] 失败: 可能是格式不对或文件损坏。系统报错: {e}")

        if not all_dfs: return [], None, {}, {}

        merged = pd.concat(all_dfs, ignore_index=True, sort=False)
        merged['SKU'] = merged['SKU'].apply(Utils.robust_sku_clean)
        merged.drop_duplicates(subset=['SKU'], inplace=True)

        sku_list = merged['SKU'].tolist()
        title_list = merged['Title'].tolist()

        family_map = defaultdict(list)
        processed_indices = set()
        for i, sku in enumerate(sku_list):
            if '-' in sku:
                root = sku.split('-')[0]
                family_map[root].append(sku)
                processed_indices.add(i)

        orphans = [i for i in range(len(sku_list)) if i not in processed_indices]
        grouped_orphans = set()
        for i in range(len(orphans)):
            idx1 = orphans[i]
            if idx1 in grouped_orphans: continue
            current_fam = [sku_list[idx1]]
            grouped_orphans.add(idx1)
            t1 = title_list[idx1]
            if len(t1) > 5:
                for j in range(i + 1, len(orphans)):
                    idx2 = orphans[j]
                    if idx2 in grouped_orphans: continue
                    if TitleAnalyzer.is_same_series(t1, title_list[idx2]):
                        current_fam.append(sku_list[idx2])
                        grouped_orphans.add(idx2)
            if len(current_fam) > 1:
                family_map[current_fam[0]].extend(current_fam)

        sku_meta = {}
        for fid, members in family_map.items():
            members = list(dict.fromkeys(members))
            if not members: continue
            leader = members[0]
            for s in members:
                sku_meta[s] = {'is_leader': (s == leader), 'members_count': len(members), 'family_members': members}

        final_family_map = {}
        for fid, members in family_map.items():
            for s in members: final_family_map[s] = fid

        # 【防呆防御】强制在导出底表前建文件夹
        base_path = Path(Config.RESULT_DIR) / "0_合并后底表.xlsx"
        base_path.parent.mkdir(parents=True, exist_ok=True)
        merged.to_excel(base_path, index=False)
        return sku_list, base_path, final_family_map, sku_meta

                df_custom = pd.read_excel(custom_title_path)
                for _, row in df_custom.iterrows():
                    sku = str(row.get('SKU', '')).strip()
                    title_en = str(row.get('Title_EN', '')).strip()
                    if sku and sku != 'nan' and title_en:
                        custom_titles[sku] = title_en
            except:
                pass

        for df in [df_base, df_img, df_erp]:
            target = None
            for c in df.columns:
                if 'sku' in str(c).lower() or '货号' in str(c).lower(): target = c; break
            if not target and len(df.columns) > 0: target = df.columns[0]
            if target:
                if target != 'SKU': df.rename(columns={target: 'SKU'}, inplace=True)
                df['SKU'] = df['SKU'].apply(Utils.robust_sku_clean)

        rename_map = {}
        for c in df_erp.columns:
            c_str = str(c)
            if 'info' in c_str or '商品信息' in c_str: rename_map[c] = '商品信息'
            if 'weight' in c_str or '重量' in c_str: rename_map[c] = '重量'
            if 'price' in c_str or '价格' in c_str: rename_map[c] = '价格'
        df_erp.rename(columns=rename_map, inplace=True)
        if '价格' in df_erp.columns:
            df_erp.sort_values(by='价格', na_position='last', inplace=True)
        df_erp.drop_duplicates(subset=['SKU'], keep='first', inplace=True)

        df_base.drop_duplicates(subset=['SKU'], inplace=True)
        df_img.drop_duplicates(subset=['SKU'], inplace=True)

        df_full = df_base.merge(df_img, on='SKU', how='left').merge(df_erp, on='SKU', how='left')

        for col in ['商品信息', '重量', '价格', 'desc_suffix', 'Size_Image', '图1']:
            if col not in df_full.columns: df_full[col] = ""
            df_full[col] = df_full[col].fillna("").astype(str)

        df_full['Parent_SKU'] = df_full['SKU'].map(family_map).fillna(df_full['SKU'])
        df_full.reset_index(drop=True, inplace=True)

        img_cols = [f'图{i}' for i in range(1, 10)]
        for col in img_cols + ['SKU图(第10列)']:
            if col not in df_full.columns: df_full[col] = ""
            df_full[col] = df_full[col].fillna("").astype(str).str.strip().replace("nan", "")

        for p_sku, group in df_full.groupby('Parent_SKU'):
            if len(group) <= 1: continue
            leader_idx = group.index[0]
            leader_imgs = [df_full.at[leader_idx, c] for c in img_cols]
            existing_imgs = set([img for img in leader_imgs if img])
            candidate_imgs = []

            for idx in group.index:
                if idx == leader_idx: continue
                child_img = df_full.at[idx, 'SKU图(第10列)']
                if child_img and (child_img not in existing_imgs) and (child_img not in candidate_imgs):
                    candidate_imgs.append(child_img)

            for i in range(1, 10):
                if not candidate_imgs: break
                col_name = f'图{i}'
                if not df_full.at[leader_idx, col_name]:
                    df_full.at[leader_idx, col_name] = candidate_imgs.pop(0)

        results = []
        for p_sku, group in df_full.groupby('Parent_SKU'):
            final_parent_title = ""
            if p_sku in custom_titles:
                final_parent_title = custom_titles[p_sku]
            else:
                for child_sku in group['SKU']:
                    if child_sku in custom_titles:
                        final_parent_title = custom_titles[child_sku]
                        break

            diff_map = {}
            if not final_parent_title:
                raw_titles = group['商品信息'].fillna('').astype(str).tolist()
                raw_titles = [t for t in raw_titles if t.lower() != 'nan' and t.strip() != '']
                if not raw_titles: raw_titles = group['Title'].replace('nan', '').dropna().astype(str).tolist()
                final_parent_title, diff_map = TitleAnalyzer.generate_smart_title_data(raw_titles)
            else:
                raw_titles = group['商品信息'].fillna('').astype(str).tolist()
                raw_titles = [t for t in raw_titles if t.lower() != 'nan' and t.strip() != '']
                _, diff_map = TitleAnalyzer.generate_smart_title_data(raw_titles)

            used_variants = defaultdict(int)

            for _, row in group.iterrows():
                raw_info = str(row['商品信息']).strip()
                var_val = diff_map.get(raw_info, "Style") if raw_info and raw_info != "nan" else "Style"
                if used_variants[var_val] > 0:
                    suffix_char = chr(65 + used_variants[var_val])
                    final_var = f"{var_val} {suffix_char}"
                else:
                    final_var = var_val
                used_variants[var_val] += 1

                cat_id, cat_attr = CategoryManager.get_info(final_parent_title)
                try:
                    w_kg = math.ceil(float(re.sub(r'[^\d.]', '', str(row['重量']))) / 10) / 100
                except:
                    w_kg = 0.01
                if w_kg < 0.01: w_kg = 0.01

                raw_imgs = [row.get(f'图{i}', '') for i in range(1, 10)]
                imgs = []
                seen_in_row = set()
                for img in raw_imgs:
                    if img and img != "nan" and img not in seen_in_row:
                        imgs.append(img)
                    seen_in_row.add(img)

                final_desc = CategoryManager.GLOBAL_DESC
                raw_desc_suffix = row.get('desc_suffix', '').strip()
                extra_desc = TranslationHelper.translate_jewelry_text(raw_desc_suffix)
                if extra_desc: final_desc += f"\n\n{extra_desc}"

                img_html = ""
                size_url = str(row.get('Size_Image', '')).strip()
                if size_url and size_url.lower() != 'nan' and size_url != "":
                    img_html += f'<br><img width="800" referrerpolicy="no-referrer" src="{size_url}"><br>\n'

                context_url = imgs[0] if imgs else ""
                if context_url and context_url != size_url:
                    img_html += f'<br><img width="800" referrerpolicy="no-referrer" src="{context_url}"><br>\n'
                final_desc += img_html

                internal_data = {
                    "cat_id": cat_id,
                    "cat_attr": cat_attr,
                    "parent_sku": p_sku,
                    "title": final_parent_title,
                    "desc": final_desc,
                    "sku": row['SKU'],
                    "var_name": final_var,
                    "var_attr_1": "style",
                    "var_val_1": final_var,
                    "price": row.get('价格', ''),
                    "stock": "9999",
                    "weight": w_kg,
                    "main_img": imgs[0] if imgs else "",
                    "img_1": imgs[1] if len(imgs) > 1 else "",
                    "img_2": imgs[2] if len(imgs) > 2 else "",
                    "img_3": imgs[3] if len(imgs) > 3 else "",
                    "img_4": imgs[4] if len(imgs) > 4 else "",
                    "img_5": imgs[5] if len(imgs) > 5 else "",
                    "img_6": imgs[6] if len(imgs) > 6 else "",
                    "img_7": imgs[7] if len(imgs) > 7 else "",
                    "img_8": imgs[8] if len(imgs) > 8 else "",
                    "var_img": row.get('SKU图(第10列)', ""),
                    "delivery_days": "2",
                    "brand": "NoBrande"
                }

                output_row = {}
                for key in TemplateManager.INTERNAL_FIELDS.keys():
                    target_header = mapping.get(key, "").strip()
                    if target_header:
                        output_row[target_header] = internal_data.get(key, "")

                results.append(output_row)

        df_res = pd.DataFrame(results)

        ordered_headers = [mapping.get(k, "").strip() for k in TemplateManager.INTERNAL_FIELDS.keys() if
                           mapping.get(k, "").strip()]
        final_df = df_res[ordered_headers] if not df_res.empty else pd.DataFrame(columns=ordered_headers)

        # 【防呆防御】强制在导出成品前建文件夹
        save_path = Path(Config.RESULT_DIR) / f"3_系统导出_{datetime.now().strftime('%H%M')}.xlsx"
        save_path.parent.mkdir(parents=True, exist_ok=True)
        final_df.to_excel(save_path, index=False)
        print(f"✅ 成功套用动态模板并生成文件: {save_path}")

def run_image_task_with_crawler(crawler, sku_list, sku_meta):
    img_path = Path(Config.RESULT_DIR) / "1_图片结果.xlsx"
    processed_skus = set()
    all_data = []

    if os.path.exists(img_path):
        try:
            df_exist = pd.read_excel(img_path)
            target = None
            for c in df_exist.columns:
                if 'sku' in str(c).lower(): target = c; break
            if target:
                df_exist[target] = df_exist[target].apply(Utils.robust_sku_clean)
                processed_skus = set(df_exist[target].tolist())
                all_data = df_exist.to_dict('records')
                print(f"📷 [图片] 检测到历史记录: 已完成 {len(processed_skus)} 个，自动跳过...")
        except:
            pass

    tasks = [s for s in sku_list if s not in processed_skus]
    if not tasks:
        print("📷 [图片] 所有任务已完成，跳过采集。")
        return

    try:
        for i, sku in enumerate(tasks):
            print(f"\n{'=' * 40}")
            print(f"📸 [图片抓取] 正在处理: {sku} ({i + 1}/{len(tasks)})")

            imgs, size_img_url = crawler.fetch_sku_images(sku)
            count = len([x for x in imgs[:9] if x])
            has_size = "✅ 有" if size_img_url else "❌ 无"

            print(f"   ✨ 状态: 完成 | 🖼️ 图片数: {count} | 📏 尺寸图: {has_size}")
            print(f"{'=' * 40}\n")

            all_data.append({
                "SKU": sku,
                **{f"图{k + 1}": v for k, v in enumerate(imgs[:9])},
                "SKU图(第10列)": imgs[9] if len(imgs) > 9 else "",
                "Size_Image": size_img_url
            })

            if i > 0 and i % 5 == 0: Utils.atomic_save(pd.DataFrame(all_data), img_path)

        Utils.atomic_save(pd.DataFrame(all_data), img_path)
    except Exception as e:
        print(f"图片任务崩溃: {e}")

    all_data = []

    if os.path.exists(erp_path):

def run_spot_check(sku_list, sku_meta):
    samples = sku_list[::3]
    if not samples:
        return True, "无有效数据可抽检"

    print(f"\n🕵️‍♂️ 启动系统防错自检：总任务 {len(sku_list)} 个，盲抽 {len(samples)} 个样本核对。")

    c_img = RinhooCrawler(sku_meta)
    c_erp = ErpCrawler()

    login_errors = []

    def img_login_wrapper():
        try:
            c_img.login()
        except Exception as e:
            login_errors.append(f"图片系统启动失败: {e}")

    def erp_login_wrapper():
        try:
            c_erp.login()
        except Exception as e:
            login_errors.append(f"ERP系统启动失败: {e}")

    t1_login = threading.Thread(target=img_login_wrapper)
    t2_login = threading.Thread(target=erp_login_wrapper)
    t1_login.start();
    t2_login.start()
    t1_login.join();
    t2_login.join()

    if login_errors:
        c_img.close()
        c_erp.close()
        return False, f"❌ 抽检阶段浏览器启动失败: {login_errors}"

    try:
        print("等待人工确认弹窗...")
        ctypes.windll.user32.MessageBoxW(0,
                                         "【抽检阶段】网页已打开。\n\n请手动处理好所有弹窗，确认干净后，点击【确定】开始核对！",
                                         "等待手动确认", 0x40040)

        img_path = Path(Config.RESULT_DIR) / "1_图片结果.xlsx"
        erp_path = Path(Config.RESULT_DIR) / "2_ERP结果.xlsx"

        try:
            df_img = pd.read_excel(img_path).set_index('SKU') if os.path.exists(img_path) else pd.DataFrame()
            df_erp = pd.read_excel(erp_path).set_index('SKU') if os.path.exists(erp_path) else pd.DataFrame()
        except Exception as e:
            return False, f"❌ 读取本地结果文件失败: {e}"

        for i, sku in enumerate(samples):
            print(f"严格抽检交叉对比: {sku} ({i + 1}/{len(samples)})")

            try:
                db_price = float(df_erp.loc[sku, 'price']) if sku in df_erp.index and pd.notna(
                    df_erp.loc[sku, 'price']) else 0.0
            except:
                db_price = 0.0

            try:
                raw_w = str(df_erp.loc[sku, '重量'] if '重量' in df_erp.columns else df_erp.loc[sku, 'weight']).replace(
                    'g', '').replace('克', '')
                db_weight = float(raw_w.strip()) if sku in df_erp.index else 0.0
            except:
                db_weight = 0.0

            db_img1 = str(df_img.loc[sku, '图1']) if sku in df_img.index and '图1' in df_img.columns else ""

            erp_info = c_erp.fetch_sku_info(sku)
            try:
                new_price = float(erp_info.get('price', 0) if erp_info and erp_info.get('price') else 0)
            except:
                new_price = 0.0

            try:
                new_weight = float(str(erp_info.get('weight', 0)).replace('g', '').replace('克',
                                                                                           '').strip() if erp_info and erp_info.get(
                    'weight') else 0)
            except:
                new_weight = 0.0

            imgs, _ = c_img.fetch_sku_images(sku)
            new_img1 = imgs[0] if imgs else ""

            if abs(new_price - db_price) > 0.01 or abs(new_weight - db_weight) > 0.01:
                return False, f"❌ [严重错位熔断] {sku} - 记录({db_price}/{db_weight}) 与 现场({new_price}/{new_weight}) 不符！"

            if new_img1 and db_img1 and new_img1 != db_img1:
                return False, f"❌ [严重错位熔断] {sku} - 首图链接错位张冠李戴！"

        return True, f"✅ 完美！抽检 {len(samples)} 个样本数据 100% 正确。"
    except Exception as e:
        return False, f"❌ 抽检过程崩溃: {str(e)}"
    finally:
        c_img.close()
        c_erp.close()

class FetchImageThread(QThread):
    finished_signal = pyqtSignal(bytes)

    def __init__(self, url):
        super().__init__()
        self.url = url

    def run(self):
        try:
            if not self.url or str(self.url).lower() == 'nan':
                self.finished_signal.emit(b"")
                return
            resp = requests.get(self.url, timeout=5, verify=False)
            if resp.status_code == 200:
                self.finished_signal.emit(resp.content)
            else:
                self.finished_signal.emit(b"")
        except:
            self.finished_signal.emit(b"")


class BatchImageFetcher(QThread):
    image_fetched = pyqtSignal(int, bytes)

    def __init__(self, fetch_list):
        super().__init__()
        self.fetch_list = fetch_list

    def run(self):
        for row, url in self.fetch_list:
            if not url or str(url).lower() == 'nan':
                self.image_fetched.emit(row, b"")
                continue
            try:
                resp = requests.get(url, timeout=5, verify=False)
                if resp.status_code == 200:
                    self.image_fetched.emit(row, resp.content)
                else:
                    self.image_fetched.emit(row, b"")
            except:
                self.image_fetched.emit(row, b"")

    def load_data(self):
        base_path = Path(Config.RESULT_DIR) / "0_合并后底表.xlsx"
        erp_path = Path(Config.RESULT_DIR) / "2_ERP结果.xlsx"
        img_path = Path(Config.RESULT_DIR) / "1_图片结果.xlsx"
        saved_path = Path(Config.RESULT_DIR) / "4_自定义标题.xlsx"

        if not base_path.exists():
            QMessageBox.warning(self, "错误", "未找到 0_合并后底表.xlsx，请先抓取！")
            return

        df_base = pd.read_excel(base_path)
        df_erp = pd.read_excel(erp_path) if erp_path.exists() else pd.DataFrame(columns=['SKU', 'info1'])
        df_img = pd.read_excel(img_path) if img_path.exists() else pd.DataFrame(columns=['SKU'])
        df_saved = pd.read_excel(saved_path) if saved_path.exists() else pd.DataFrame(columns=['SKU', 'Title_EN'])

        sku_cn_map = {}
        for _, row in df_base.iterrows():
            sku = str(row.get('SKU', '')).strip()
            title = str(row.get('Title', '')).strip()
            if sku and sku != 'nan': sku_cn_map[sku] = title

        for _, row in df_erp.iterrows():
            sku = str(row.get('SKU', '')).strip()
            info1 = str(row.get('info1', '')).strip()
            if sku and sku != 'nan' and info1 and info1 != 'nan':
                sku_cn_map[sku] = info1

        sku_img_map = {}
        if not df_img.empty:
            target = None
            for c in df_img.columns:
                if 'sku' in str(c).lower(): target = c; break
            if target:
                for _, row in df_img.iterrows():
                    sku = str(row[target]).strip()
                    var_img = str(row.get('SKU图(第10列)', '')).strip()
                    if not var_img or var_img.lower() == 'nan':
                        var_img = str(row.get('图1', '')).strip()
                    if var_img and var_img.lower() != 'nan':
                        sku_img_map[sku] = var_img

        sku_en_map = {}
        for _, row in df_saved.iterrows():
            sku = str(row.get('SKU', '')).strip()
            en = str(row.get('Title_EN', '')).strip()
            if sku and sku != 'nan': sku_en_map[sku] = en

        self.tab.setRowCount(len(sku_cn_map))
        fetch_list = []

        for i, (sku, cn_title) in enumerate(sku_cn_map.items()):
            en_title = sku_en_map.get(sku, "")

            chk_item = QTableWidgetItem()
            chk_item.setFlags(Qt.ItemIsUserCheckable | Qt.ItemIsEnabled)
            chk_item.setCheckState(Qt.Unchecked)
            self.tab.setItem(i, 0, chk_item)

            img_lbl = QLabel("加载中...")
            img_lbl.setAlignment(Qt.AlignCenter)
            self.tab.setCellWidget(i, 1, img_lbl)

            url = sku_img_map.get(sku, "")
            if url:
                fetch_list.append((i, url))
            else:
                img_lbl.setText("无图")

            self.tab.setItem(i, 2, QTableWidgetItem(sku))
            self.tab.setItem(i, 3, QTableWidgetItem(cn_title))
            self.tab.setItem(i, 4, QTableWidgetItem(en_title))
            self.tab.setItem(i, 5, QTableWidgetItem("✅ 有翻译" if en_title else "⏳ 待匹配"))
            if en_title:
                self.tab.item(i, 4).setBackground(Qt.green)

        if fetch_list:
            self.fetcher = BatchImageFetcher(fetch_list)
            self.fetcher.image_fetched.connect(self.on_image_fetched)
            self.fetcher.start()

        self.main_window.log_box.append("✔️ 标题匹配数据与图片挂载任务已启动！")

            titles.append(self.tab.item(r, 3).text().strip())

        if titles:
            QApplication.clipboard().setText("\n".join(titles))
            QMessageBox.information(self, "复制成功",

    def open_match_dialog(self):
        if self.tab.rowCount() == 0:
            QMessageBox.warning(self, "错误", "请先加载数据！")
            return

        d = QDialog(self)
        d.setWindowTitle("粘贴标题匹配 (终极智能捕捉版)")
        d.resize(550, 400)
        v = QVBoxLayout(d)
        v.addWidget(QLabel("直接粘贴您的内容（无视特殊符号或换行，只要包含中文即可100%匹配提取翻译）:"))
        te = QTextEdit()
        te.setAcceptRichText(False)
        v.addWidget(te)
        btn = QPushButton("开启智能捕捉并覆盖表格")
        v.addWidget(btn)

        def run_match():
            raw = te.toPlainText().strip()
            if not raw: return

            match_count = 0
            lines = raw.split('\n')
            row_matchers = []

            for r in range(self.tab.rowCount()):
                table_cn = self.tab.item(r, 3).text().strip()
                chars = re.findall(r'[\w\u4e00-\u9fa5]', table_cn)
                if chars:
                    pattern = r'^\W*' + r'\W*'.join([re.escape(c) for c in chars]) + r'\W*'
                    regex = re.compile(pattern, re.IGNORECASE)
                    row_matchers.append((r, regex))

            for l in lines:
                l = l.strip()
                if not l: continue
                for r, regex in row_matchers:
                    match = regex.match(l)
                    if match:
                        en = l[match.end():].strip()
                        if en:
                            self.tab.setItem(r, 4, QTableWidgetItem(en))
                            self.tab.item(r, 4).setBackground(Qt.green)
                            self.tab.setItem(r, 5, QTableWidgetItem("✨ 新匹配"))
                            match_count += 1
                        break

            d.accept()
            if match_count > 0:
                QMessageBox.information(self, "匹配成功",
                                        f"✅ 智能捕捉完成！成功填入 {match_count} 条数据！\n记得点击【保存匹配结果】！")
            else:
                QMessageBox.warning(self, "匹配失败", "⚠️ 匹配结果为 0 条！")

        btn.clicked.connect(run_match)
        d.exec_()

    def save_data(self):
        if self.tab.rowCount() == 0: return
        data = []
        for r in range(self.tab.rowCount()):
            sku = self.tab.item(r, 2).text().strip()
            en = self.tab.item(r, 4).text().strip() if self.tab.item(r, 4) else ""
            if en: data.append({"SKU": sku, "Title_EN": en})

        if data:
            save_path = Path(Config.RESULT_DIR) / "4_自定义标题.xlsx"
            Utils.atomic_save(pd.DataFrame(data), save_path)
            self.main_window.log_box.append(f"💾 成功保存 {len(data)} 个自定义外语标题！")
            QMessageBox.information(self, "成功", "保存成功！去生成导入表格时会自动采用这里的标题。")

            self.family_map.clear()
            for _, row in df_base.iterrows():
                sku = str(row['SKU']).strip()
                parent = sku.split('-')[0] if '-' in sku else sku
                if parent not in self.family_map: self.family_map[parent] = []
                self.family_map[parent].append(sku)

            self.df_img = pd.read_excel(img_path)
            target = None
            for c in self.df_img.columns:
                if 'sku' in str(c).lower(): target = c; break
            if target and target != 'SKU': self.df_img.rename(columns={target: 'SKU'}, inplace=True)
            self.df_img['SKU'] = self.df_img['SKU'].astype(str).str.strip()
            self.df_img.set_index('SKU', inplace=True)

            self.list_parents.clear()
            parents = sorted(list(self.family_map.keys()))
            self.list_parents.addItems(parents)

            self.main_window.log_box.append("✔️ 图片数据已成功加载到预览页！")
            QMessageBox.information(self, "成功", "最新图片数据加载成功！")
        except Exception as e:
            QMessageBox.critical(self, "加载失败", f"错误信息: {e}")

    def on_parent_selected(self):
        items = self.list_parents.selectedItems()
        if not items: return
        parent = items[0].text()

        self.list_children.blockSignals(True)
        self.list_children.clear()
        children = self.family_map.get(parent, [])

        fetch_list = []
        for i, child_sku in enumerate(children):
            item = QListWidgetItem(child_sku)
            self.list_children.addItem(item)

            url = ""
            if child_sku in self.df_img.index:
                row = self.df_img.loc[child_sku]
                url = str(row.get('SKU图(第10列)', '')).strip()
                if not url or url.lower() == 'nan':
                    url = str(row.get('图1', '')).strip()
            if url and url.lower() != 'nan':
                fetch_list.append((i, url))

        self.list_children.blockSignals(False)

        if fetch_list:
            self.fetcher = BatchImageFetcher(fetch_list)
            self.fetcher.image_fetched.connect(self.on_child_icon_fetched)
            self.fetcher.start()

        if children:
            self.list_children.setCurrentRow(0)
            self.on_child_selected()

    def on_child_icon_fetched(self, row, data):
        if data and row < self.list_children.count():
            item = self.list_children.item(row)
            if item:
                pixmap = QPixmap()
                pixmap.loadFromData(data)
                icon = QIcon(pixmap.scaled(60, 60, Qt.KeepAspectRatioByExpanding, Qt.SmoothTransformation))
                item.setIcon(icon)

    def on_child_selected(self):
        items = self.list_children.selectedItems()
        if not items: return
        child_sku = items[0].text()
        self.current_child_sku = child_sku
        self.lbl_current_sku.setText(f"当前精修变种: {child_sku}")

        if child_sku in self.df_img.index:
            row = self.df_img.loc[child_sku]
            for col, slot in self.image_slots.items():
                url = row.get(col, "") if col in self.df_img.columns else ""
                slot.set_data(url)
        else:
            for slot in self.image_slots.values():
                slot.set_data("")

    def shift_images_forward(self, deleted_col):
        if deleted_col == "SKU图(第10列)":
            self.image_slots[deleted_col].set_data("")
            self.main_window.log_box.append(f"🔄 变种缩略图已清空！请记得点击[覆盖保存]。")
            return

        main_cols = [f"图{i}" for i in range(1, 10)]
        new_urls = []
        for col in main_cols:
            if col == deleted_col: continue
            url = self.image_slots[col].txt_url.text().strip()
            if url: new_urls.append(url)

        while len(new_urls) < 9: new_urls.append("")

        for i, col in enumerate(main_cols):
            self.image_slots[col].set_data(new_urls[i])

        self.main_window.log_box.append(f"🔄 主图已自动填补空位并前移（变种图位置保持不变）！请记得点击[覆盖保存]。")

    def save_current_sku(self):
        if not self.current_child_sku or self.df_img is None: return

        if self.current_child_sku not in self.df_img.index:
            self.df_img.loc[self.current_child_sku] = ""

        for col, slot in self.image_slots.items():
            self.df_img.at[self.current_child_sku, col] = slot.txt_url.text().strip()

        try:
            img_path = Path(Config.RESULT_DIR) / "1_图片结果.xlsx"
            df_to_save = self.df_img.reset_index()
            Utils.atomic_save(df_to_save, img_path)
            self.main_window.log_box.append(f"💾 变种 {self.current_child_sku} 已经覆盖保存至 1_图片结果.xlsx！")
            QMessageBox.information(self, "成功", f"变种 {self.current_child_sku} 的图片顺序与修改已永久保存！")
        except Exception as e:
            QMessageBox.critical(self, "保存失败", f"错误信息: {e}")

    def generate_final(self):
        self.main_window.trigger_merge_only()

    def load_existing_mapping(self):
        mapping = TemplateManager.get_current_mapping()
        self.external_headers = list(set(mapping.values()))
        if "" in self.external_headers: self.external_headers.remove("")

        for i, (int_key, (disp_name, is_req)) in enumerate(TemplateManager.INTERNAL_FIELDS.items()):
            display_text = f"* {disp_name}" if is_req else disp_name
            item = QTableWidgetItem(display_text)
            if is_req: item.setForeground(Qt.red)
            self.map_table.setItem(i, 0, item)

            combo = QComboBox()
            combo.addItem("—— 留空不填(系统自动过滤) ——")
            combo.addItems(self.external_headers)

            saved_val = mapping.get(int_key, "")
            if saved_val:
                idx = combo.findText(saved_val)
                if idx >= 0: combo.setCurrentIndex(idx)

            self.map_table.setCellWidget(i, 1, combo)
            self.map_table.setItem(i, 2, QTableWidgetItem("✔️ 已加载预设" if saved_val else "——"))

    def select_template(self):
        path, _ = QFileDialog.getOpenFileName(self, "选择空白模板文件", "", "Excel Files (*.xlsx *.xls *.csv)")
        if path:
            self.template_path = path
            self.preview_header()

    def preview_header(self):
        if not self.template_path: return
        row_idx = self.spin_row.value() - 1
        try:
            if self.template_path.endswith('.csv'):
                df = pd.read_csv(self.template_path, header=row_idx)
            else:
                df = pd.read_excel(self.template_path, header=row_idx)
            self.external_headers = [str(c) for c in df.columns if not str(c).startswith('Unnamed:')]
            preview_text = " | ".join(self.external_headers[:5]) + " ..."
            self.lbl_preview.setText(f"预览: {preview_text}")
        except Exception as e:
            self.lbl_preview.setText("读取失败，请检查行数或文件格式")

    def auto_match(self):
        if not self.external_headers:
            QMessageBox.warning(self, "警告", "请先上传并预览正确的外部模板！")
            return

        for i, (int_key, (disp_name, is_req)) in enumerate(TemplateManager.INTERNAL_FIELDS.items()):
            combo = QComboBox()
            combo.addItem("—— 留空不填(系统自动过滤) ——")
            combo.addItems(self.external_headers)

            best_match = ""
            best_score = 0
            keywords = TemplateManager.FUZZY_DICT.get(int_key, [disp_name])

            for ext_h in self.external_headers:
                clean_ext = ext_h.lower()
                for kw in keywords:
                    ratio = difflib.SequenceMatcher(None, kw.lower(), clean_ext).ratio()
                    if kw.lower() in clean_ext: ratio += 0.4
                    if ratio > best_score and ratio > 0.5:
                        best_score = ratio
                        best_match = ext_h

            if best_match:
                idx = combo.findText(best_match)
                if idx >= 0: combo.setCurrentIndex(idx)

            self.map_table.setCellWidget(i, 1, combo)

            status_item = QTableWidgetItem("✨ AI自动匹配" if best_match else "⏳ 未匹配")
            if best_match: status_item.setForeground(Qt.blue)
            self.map_table.setItem(i, 2, status_item)

        QMessageBox.information(self, "识别完成", "AI 智能连线完毕！请检查下拉框，可做人工微调修正。")

    def save_mapping(self):
        new_mapping = {}
        missing_required = []

        for i, (int_key, (disp_name, is_req)) in enumerate(TemplateManager.INTERNAL_FIELDS.items()):
            combo = self.map_table.cellWidget(i, 1)
            selected = combo.currentText()

            if selected.startswith("——"):
                new_mapping[int_key] = ""
                if is_req: missing_required.append(disp_name)
            else:
                new_mapping[int_key] = selected

        if missing_required:
            msg = "⛔ 【防呆拦截】以下核心必填字段被您留空了：\n\n" + "\n".join(
                missing_required) + "\n\n请在下拉框中为它们指定对应的外部表头！"
            QMessageBox.critical(self, "配置错误", msg)
            return

        TemplateManager.save_mapping(new_mapping)
        self.main_window.log_box.append("💾 动态模板配置已永久保存！")
        QMessageBox.information(self, "保存成功", "✅ 模板配置已生效！下次生成最终表格时，将自动使用该表头结构。")

class EmittingStream(QObject):
    textWritten = pyqtSignal(str)

    def write(self, text):
        if str(text).strip(): self.textWritten.emit(str(text))

    def flush(self): pass
